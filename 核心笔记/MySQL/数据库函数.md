```mysql
数据库函数
1.字符串函数
-- concat（连接字符串）
select concat ('hello',' mysql');

-- lower（变为小写）
select lower('HELLO');

-- upper（变为大写）
select upper('hello');

-- lpad（左填充）
select lpad('01',5,'-');

-- rpad（右填充）
select rpad('01',5,'-');

-- trim（去除首尾空格）
select trim('    hello  mysql   ');

-- substrig（截取字符串，从1开始）
select substring('hello mysql',1,7);

```

```mysql
2.数值函数
-- ceil（向上取整）
select ceil(1.1);

-- floor（向下取整）
select floor(1.9);

-- mod（取模运算）
select mod(5,3);

-- rand（随机数0~1）
select rand();

-- round（四舍五入保留小数）
select  round(2.345,2);
```



```mysql
3.日期函数
-- curdate() （获取当前日期）
select  curdate();

-- curtime() （获取当前时间）
select curtime();


-- now() （获取当前日期和时间）
select now();

-- year month day（获取指定的年、月、日）
select year(now());
select  month(now());
select day(now());

-- date_add （在当前时间的基础上添加指定的时间）
select date_add(now(), interval  70 month);

-- datediff （两个日期相差多少天，左边减右边）
select datediff('2020-12-01','2021-10-01');
```



```mysql
4.流程函数
-- if （判断第一个表达式是否为true,是返回第一个参数，否则）
select if(true,'ok','error');

-- ifnull（判断第一个表达式是否为null,是null返回第二个参数，否则）
select ifnull('ok','default');

select ifnull('','default');

select ifnull(null,'default');


-- case when else end（进行条件分支的判断）
select name,
       ( case workaddress when '北京'   then '一线城市' when '上海' then '一线城市' else   '二线城市' end) as '工作地址'
from emp;

```
